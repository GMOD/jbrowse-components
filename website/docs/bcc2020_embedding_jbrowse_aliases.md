const fetch = require('node-fetch')

async function main() {
  const server = 'https://rest.ensembl.org'
  const ext = '/info/assembly/homo_sapiens'
  const params = 'synonyms=1'
  let response
  try {
    response = await fetch(`${server}${ext}?${params}`, {
      headers: {
        'Content-Type': 'application/json',
      },
    })
  } catch (error) {
    throw new Error(error)
  }
  if (!response.ok) {
    throw new Error(
      `Network response failure: ${response.status} (${response.statusText})`,
    )
  }
  let assemblyInfo
  try {
    assemblyInfo = await response.json()
  } catch (error) {
    throw new Error(error)
  }
  assemblyInfo.top_level_region
    .sort(({ name: a }, { name: b }) => {
      if (a < b) {
        return -1
      }
      if (a > b) {
        return 1
      }
      return 0
    })
    .forEach(region => {
      const synonyms =
        region.synonyms && region.synonyms.map(synonym => synonym.name)
      if (synonyms && synonyms.length) {
        console.log(`${region.name}\t${synonyms.join('\t')}`)
      }
    })
}
